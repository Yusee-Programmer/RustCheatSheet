1️⃣ TRAITS (Defining and Implementing)
📖 What is it?

Traits define shared behavior that types can implement. They're Rust's version of interfaces but more powerful, allowing method definitions with default implementations, associated functions, supertraits, and blanket implementations.
🎯 When to Use Traits

    Defining common behavior: When multiple types share functionality

    Creating abstractions: When you want to write generic code

    Extending types: Adding methods to types you don't own

    Polymorphism: Static and dynamic dispatch

💡 Why Use Traits

    Abstraction: Define behavior without concrete types

    Reusability: Write code once for many types

    Safety: Compile-time guarantees about behavior

    Flexibility: Default implementations, inheritance, blanket impls

📝 Complete Methods Reference
Defining Traits
rust

// Why traits: Define shared behavior
// When: Multiple types need to implement same methods

// Basic trait definition
trait Greet {
    // Required method (no default)
    fn greet(&self) -> String;
    
    // Method with default implementation
    fn farewell(&self) -> String {
        "Goodbye!".to_string()
    }
    
    // Method that uses other methods
    fn introduce(&self) -> String {
        format!("{} - {}", self.greet(), self.farewell())
    }
    
    // Associated function (static method)
    fn default_greeting() -> String {
        "Hello".to_string()
    }
}

// Trait with associated types
trait Container {
    type Item;  // Associated type
    fn get(&self) -> &Self::Item;
    fn set(&mut self, item: Self::Item);
}

// Trait inheritance (supertrait)
trait Animal {
    fn name(&self) -> &str;
    fn sound(&self) -> String;
}

trait Pet: Animal {  // Pet requires Animal
    fn owner(&self) -> &str;
    
    // Default implementation using supertrait methods
    fn describe(&self) -> String {
        format!("{} says {} and belongs to {}", 
                self.name(), self.sound(), self.owner())
    }
}

// Marker trait (no methods)
trait ThreadSafe {}
trait Encryptable {}

Implementing Traits
rust

// Why implement: Provide behavior for types
// When: Adding trait functionality to types

// Basic implementation
struct Person {
    name: String,
    age: u32,
}

impl Greet for Person {
    fn greet(&self) -> String {
        format!("Hello, I'm {}", self.name)
    }
    
    // Override default implementation
    fn farewell(&self) -> String {
        format!("Goodbye from {}!", self.name)
    }
}

// Implementation with associated type
struct BoxContainer<T> {
    value: T,
}

impl<T> Container for BoxContainer<T> {
    type Item = T;
    
    fn get(&self) -> &Self::Item {
        &self.value
    }
    
    fn set(&mut self, item: Self::Item) {
        self.value = item;
    }
}

// Implementation for multiple traits
struct Dog {
    name: String,
    owner_name: String,
}

impl Animal for Dog {
    fn name(&self) -> &str {
        &self.name
    }
    
    fn sound(&self) -> String {
        "Woof!".to_string()
    }
}

impl Pet for Dog {
    fn owner(&self) -> &str {
        &self.owner_name
    }
    // describe() uses default implementation
}

// Implementation with where clause
impl<T: Clone + Debug> Container for BoxContainer<T> {
    type Item = T;
    
    fn get(&self) -> &T {
        &self.value
    }
    
    fn set(&mut self, item: T) {
        self.value = item;
    }
}

Default Implementations
rust

// Why default: Reduce boilerplate
// When: Most implementations share common behavior

trait Logger {
    // Must be implemented
    fn log(&self, msg: &str);
    
    // Default implementations (can be overridden)
    fn info(&self, msg: &str) {
        self.log(&format!("INFO: {}", msg));
    }
    
    fn error(&self, msg: &str) {
        self.log(&format!("ERROR: {}", msg));
    }
    
    fn warn(&self, msg: &str) {
        self.log(&format!("WARN: {}", msg));
    }
    
    // Default using other trait methods
    fn debug(&self, msg: &str) {
        #[cfg(debug_assertions)]
        self.log(&format!("DEBUG: {}", msg));
    }
}

// Console logger uses all defaults
struct ConsoleLogger;

impl Logger for ConsoleLogger {
    fn log(&self, msg: &str) {
        println!("{}", msg);
    }
}

// File logger overrides some defaults
struct FileLogger {
    file: std::fs::File,
}

impl Logger for FileLogger {
    fn log(&self, msg: &str) {
        // Write to file...
    }
    
    fn error(&self, msg: &str) {
        // Special error formatting
        self.log(&format!("❌ ERROR: {}", msg));
    }
}

Derivable Traits (Built-in)
rust

// Why derive: Automatic implementation of common traits
// When: Types need standard behavior

// Debug - enables {:?} formatting
#[derive(Debug)]
struct Point {
    x: i32,
    y: i32,
}

// Clone - allows explicit cloning
#[derive(Clone)]
struct Config {
    host: String,
    port: u16,
}

// Copy - allows implicit copying (requires Clone)
#[derive(Copy, Clone)]
struct Coordinate {
    x: i32,
    y: i32,
}

// PartialEq - enables == and !=
#[derive(PartialEq)]
struct Color {
    r: u8,
    g: u8,
    b: u8,
}

// Eq - for strict equality (requires PartialEq)
#[derive(PartialEq, Eq)]
struct Id(u32);

// PartialOrd - enables <, >, <=, >=
#[derive(PartialOrd, PartialEq)]
struct Score(u32);

// Ord - for total ordering (requires Eq + PartialOrd)
#[derive(Ord, PartialOrd, Eq, PartialEq)]
struct Priority(i32);

// Default - provides default values
#[derive(Default)]
struct Settings {
    volume: u8,      // Default: 0
    brightness: u8,  // Default: 0
    theme: String,   // Default: ""
    enabled: bool,   // Default: false
}

// Hash - for use in HashMaps
use std::hash::Hash;
#[derive(Hash, PartialEq, Eq)]
struct UserId(u64);

// Combined example
#[derive(Debug, Clone, PartialEq, Default)]
struct User {
    username: String,
    email: String,
    age: u32,
}

fn derive_example() {
    let user1 = User::default();
    let user2 = user1.clone();
    
    println!("Debug: {:?}", user1);
    
    if user1 == user2 {
        println!("Users are equal!");
    }
}

Supertraits (Trait Inheritance)
rust

// Why supertraits: Build trait hierarchies
// When: One trait logically depends on another

trait Drawable {
    fn draw(&self);
}

trait Animated: Drawable {  // Animated requires Drawable
    fn animate(&self) {
        println!("Animating...");
        self.draw();  // Can use Drawable methods
    }
}

// Must implement both traits
struct Circle {
    radius: f64,
}

impl Drawable for Circle {
    fn draw(&self) {
        println!("Drawing circle with radius {}", self.radius);
    }
}

impl Animated for Circle {}  // Uses default animate()

// Multiple supertraits
trait Serializable: serde::Serialize + serde::de::DeserializeOwned {
    fn to_json(&self) -> String {
        serde_json::to_string(self).unwrap()
    }
    
    fn from_json(json: &str) -> Result<Self, serde_json::Error>
    where
        Self: Sized,
    {
        serde_json::from_str(json)
    }
}

// Using where clause for supertrait bounds
trait Printable
where
    Self: std::fmt::Debug + std::fmt::Display,
{
    fn print(&self) {
        println!("Display: {}, Debug: {:?}", self, self);
    }
}

Blanket Implementations
rust

// Why blanket: Implement trait for all types with certain bounds
// When: Behavior applies universally

use std::fmt::Debug;

// Blanket implementation for all types that implement Debug
trait Describe {
    fn describe(&self) -> String;
}

impl<T: Debug> Describe for T {
    fn describe(&self) -> String {
        format!("{:?}", self)
    }
}

// Blanket implementation for all types that implement Display
use std::fmt::Display;

impl<T: Display> Describe for T {
    default fn describe(&self) -> String {
        format!("{}", self)
    }
}

// Standard library example: ToString for all Display types
// impl<T: Display> ToString for T {
//     fn to_string(&self) -> String { ... }
// }

// Another example: Into for all From implementations
// impl<T, U> Into<U> for T where U: From<T> {
//     fn into(self) -> U { U::from(self) }
// }

// Blanket for all types implementing Iterator
trait Collectable {
    fn collect_vec(self) -> Vec<Self::Item>
    where
        Self: Iterator + Sized,
        Self::Item: Clone,
    {
        self.collect()
    }
}

impl<I: Iterator> Collectable for I {}

Marker Traits
rust

// Why marker: Signal capabilities without methods
// When: Types have compile-time properties

// Standard marker traits
// Send: Safe to send between threads
// Sync: Safe to share between threads
// Copy: Can be copied implicitly
// Sized: Known size at compile time
// Unpin: Can be safely moved after pinning

// Custom marker trait
trait ThreadSafe {}

// Implement for types that are thread-safe
impl ThreadSafe for i32 {}
impl ThreadSafe for String {}
impl<T: Send + Sync> ThreadSafe for Vec<T> {}

// Not thread-safe types don't implement
struct UnsafeCell<T>(std::cell::UnsafeCell<T>);
// UnsafeCell does NOT implement ThreadSafe

// Using marker trait as bound
fn process<T: ThreadSafe>(value: T) {
    println!("Value is thread-safe!");
}

// Conditional compilation with markers
fn marker_example() {
    process(42);  // ✅ i32 is thread-safe
    process("hello".to_string());  // ✅ String is thread-safe
    
    // process(UnsafeCell::new(5));  // ❌ Not thread-safe
}

Orphan Rule and Newtype Pattern
rust

// Why orphan rule: Prevent conflicts between crates
// When: Implementing external traits on external types

use std::fmt::Display;

// ❌ Cannot implement Display for Vec (both external)
// impl<T> Display for Vec<T> { }  // Error: orphan rule

// ✅ Newtype pattern - wrap the external type
struct PrettyVec<T>(Vec<T>);

impl<T: Display> Display for PrettyVec<T> {
    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
        write!(f, "[")?;
        for (i, item) in self.0.iter().enumerate() {
            if i > 0 {
                write!(f, ", ")?;
            }
            write!(f, "{}", item)?;
        }
        write!(f, "]")
    }
}

// Multiple newtype wrappers for different formatting
struct JsonVec<T>(Vec<T>);
struct XmlVec<T>(Vec<T>);

// Newtype for strong typing
struct UserId(u64);
struct ProductId(u64);

fn newtype_example() {
    let v = PrettyVec(vec![1, 2, 3]);
    println!("{}", v);  // "[1, 2, 3]"
    
    let user = UserId(42);
    let product = ProductId(42);
    
    // Can't mix types!
    // let id: u64 = user;  // ❌ Can't assign
    // let same = user == product;  // ❌ Can't compare
}

🏆 Best Practices
rust

// 1. Provide default implementations when possible
trait Logger {
    fn log(&self, msg: &str);
    
    fn error(&self, msg: &str) {
        self.log(&format!("ERROR: {}", msg));
    }
}

// 2. Use supertraits for hierarchies
trait Drawable { }
trait Animated: Drawable { }

// 3. Derive common traits
#[derive(Debug, Clone, PartialEq)]
struct MyType { }

// 4. Use marker traits for capabilities
trait Readable { }
trait Writable: Readable { }

// 5. Document trait requirements
/// A trait for types that can be hashed.
/// 
/// # Requirements
/// The type must implement `Eq` and `Hash`.
trait Hashable: Eq + std::hash::Hash { }

// 6. Use where clause for complex bounds
trait Complex where Self: Clone + Debug + Default { }

// 7. Use associated types for single output types
trait Iterator {
    type Item;
    fn next(&mut self) -> Option<Self::Item>;
}

// 8. Keep traits focused (single responsibility)
trait Serialize { }  // Good
trait SerializeAndDeserialize { }  // Bad (two responsibilities)

2️⃣ TRAIT BOUNDS
📖 What is it?

Trait bounds specify that a generic type must implement certain traits. They enable compile-time polymorphism and type-safe generic code.
🎯 When to Use Trait Bounds

    Generic functions: When you need to call trait methods

    Generic structs: When fields need specific capabilities

    Conditional implementations: When methods only available for types with certain traits

    API design: When documenting requirements

💡 Why Use Trait Bounds

    Safety: Compiler ensures only valid types are used

    Flexibility: Write once, work with many types

    Clarity: Document what capabilities a generic type needs

    Performance: Zero-cost abstraction (monomorphization)

📝 Complete Methods Reference
Basic Trait Bounds
rust

// Why bounds: Constrain generic types
// When: Generic code needs specific capabilities

use std::fmt::{Debug, Display};

// Single bound
fn print_debug<T: Debug>(value: T) {
    println!("{:?}", value);
}

// Multiple bounds with + syntax
fn print_both<T: Debug + Display>(value: T) {
    println!("Display: {}", value);
    println!("Debug: {:?}", value);
}

// Different bounds for different parameters
fn compare_and_print<T: PartialOrd, U: Debug>(a: T, b: T, debug: U) {
    if a > b {
        println!("a is greater");
    }
    println!("Debug: {:?}", debug);
}

// Bound on generic struct
struct Wrapper<T: Debug> {
    value: T,
}

impl<T: Debug> Wrapper<T> {
    fn print(&self) {
        println!("{:?}", self.value);
    }
}

Where Clauses (Cleaner Syntax)
rust

// Why where: Improve readability for complex bounds
// When: Multiple bounds or bounds on multiple types

// Without where (hard to read)
fn complex_function<T, U, V>(a: T, b: U, c: V) -> T
where
    T: Display + Clone + Debug,
    U: PartialEq + Debug,
    V: Clone + Default,
{
    println!("a: {}", a);
    println!("b: {:?}", b);
    let cloned = c.clone();
    a
}

// With where (clean and readable)
fn complex_function<T, U, V>(a: T, b: U, c: V) -> T
where
    T: Display + Clone + Debug,
    U: PartialEq + Debug,
    V: Clone + Default,
{
    // Same implementation
    a
}

// Struct with where clause
struct Pair<T, U>
where
    T: Debug + Clone,
    U: Debug + PartialEq,
{
    first: T,
    second: U,
}

impl<T, U> Pair<T, U>
where
    T: Debug + Clone,
    U: Debug + PartialEq,
{
    fn new(first: T, second: U) -> Self {
        Pair { first, second }
    }
    
    fn print(&self) {
        println!("First: {:?}, Second: {:?}", self.first, self.second);
    }
}

Return Position impl Trait
rust

// Why impl Trait: Return opaque types
// When: You don't want to expose concrete type

// Returns some type that implements Debug
fn make_debuggable(flag: bool) -> impl Debug {
    if flag {
        42
    } else {
        "hello"
    }
} // ❌ Won't compile - returns different types

// ✅ Returns single concrete type
fn make_integer() -> impl Debug {
    42
}

// ✅ Returns closure
fn make_adder(x: i32) -> impl Fn(i32) -> i32 {
    move |y| x + y
}

// ✅ Returns iterator
fn make_range(start: i32, end: i32) -> impl Iterator<Item = i32> {
    start..end
}

// ✅ Returns iterator with adaptors
fn make_squares(limit: i32) -> impl Iterator<Item = i32> {
    (1..=limit).map(|x| x * x)
}

// Usage
fn impl_trait_example() {
    let add5 = make_adder(5);
    println!("{}", add5(10));  // 15
    
    for num in make_range(1, 5) {
        println!("{}", num);  // 1, 2, 3, 4
    }
    
    for square in make_squares(5) {
        println!("{}", square);  // 1, 4, 9, 16, 25
    }
}

// In struct fields (requires GATs or boxing)
struct Container {
    // iter: impl Iterator<Item = i32>,  // ❌ Not allowed in struct fields
    iter: Box<dyn Iterator<Item = i32>>,  // ✅ Use trait object instead
}

Common Standard Library Bounds
rust

// Why common bounds: Reuse standard library functionality
// When: Implementing common patterns

use std::fmt::Debug;

// Clone and Debug
fn clone_and_print<T: Clone + Debug>(value: T) {
    let cloned = value.clone();
    println!("Original: {:?}, Cloned: {:?}", value, cloned);
}

// PartialOrd for comparisons
fn max<T: PartialOrd>(a: T, b: T) -> T {
    if a > b { a } else { b }
}

// Iterator and IntoIterator
fn sum_all<I>(iterable: I) -> i32
where
    I: IntoIterator<Item = i32>,
{
    iterable.into_iter().sum()
}

// AsRef for cheap conversions
fn read_config<P: AsRef<std::path::Path>>(path: P) -> String {
    std::fs::read_to_string(path.as_ref()).unwrap()
}

// From and Into
fn convert_to_string<T: Into<String>>(value: T) -> String {
    value.into()
}

// Fn traits for closures
fn apply_twice<F>(mut f: F, value: i32) -> i32
where
    F: FnMut(i32) -> i32,
{
    let first = f(value);
    f(first)
}

// Default for fallback values
fn with_default<T: Default>(value: Option<T>) -> T {
    value.unwrap_or_default()
}

// Usage
fn common_bounds_example() {
    clone_and_print(42);
    println!("Max: {}", max(10, 20));
    println!("Sum: {}", sum_all(vec![1, 2, 3]));
    
    let doubled = apply_twice(|x| x * 2, 5);
    println!("Doubled twice: {}", doubled);  // 20
    
    let value = with_default(Some(42));
    println!("Value: {}", value);  // 42
    
    let default = with_default(None::<i32>);
    println!("Default: {}", default);  // 0
}

Higher-Ranked Trait Bounds (HRTB)
rust

// Why HRTB: "For all lifetimes" constraints
// When: Closures or functions that work with any lifetime

// HRTB: for any lifetime 'a
fn call_with_ref<F>(f: F) -> i32
where
    F: for<'a> Fn(&'a i32) -> i32,
{
    let x = 42;
    f(&x)
}

// Without HRTB, this wouldn't work
fn call_with_ref_generic<F, T>(f: F, value: T) -> T
where
    F: Fn(&T) -> T,
{
    f(&value)
}

// Usage
fn hrtb_example() {
    let result = call_with_ref(|&x| x * 2);
    println!("{}", result);  // 84
    
    // Works with references
    let s = String::from("hello");
    let result = call_with_ref_generic(|x: &String| x.len(), s);
    println!("{}", result);  // 5
}

// HRTB with trait objects
trait Callback {
    fn call(&self, x: &i32) -> i32;
}

impl<F> Callback for F
where
    F: for<'a> Fn(&'a i32) -> i32,
{
    fn call(&self, x: &i32) -> i32 {
        self(x)
    }
}

// Common in iterator adaptors
fn process_iter<I>(iter: I)
where
    I: for<'a> Iterator<Item = &'a i32>,
{
    for item in iter {
        println!("{}", item);
    }
}

Conditional Implementations
rust

// Why conditional: Implement methods only for types with certain traits
// When: Some methods require additional capabilities

struct Container<T> {
    value: T,
}

// Methods for all T
impl<T> Container<T> {
    fn new(value: T) -> Self {
        Container { value }
    }
    
    fn get(&self) -> &T {
        &self.value
    }
}

// Methods only for T that implement Debug
impl<T: Debug> Container<T> {
    fn debug_print(&self) {
        println!("{:?}", self.value);
    }
}

// Methods only for T that implement Clone
impl<T: Clone> Container<T> {
    fn cloned(&self) -> Self {
        Container {
            value: self.value.clone(),
        }
    }
}

// Methods only for T that implement PartialOrd
impl<T: PartialOrd> Container<T> {
    fn max(&self, other: &Self) -> &Self {
        if self.value > other.value {
            self
        } else {
            other
        }
    }
}

// Usage
fn conditional_impl_example() {
    let c1 = Container::new(42);
    c1.debug_print();  // Works: i32 implements Debug
    
    let c2 = Container::new("hello".to_string());
    c2.debug_print();  // Works: String implements Debug
    
    let c3 = Container::new(vec![1, 2, 3]);
    // c3.debug_print();  // ❌ Vec doesn't implement Debug (unless derived)
    
    let cloned = c1.cloned();  // Works: i32 implements Clone
    // let cloned = c3.cloned();  // ❌ Vec doesn't implement Clone (unless derived)
}

🏆 Best Practices
rust

// 1. Prefer where clauses for complex bounds
fn complex<T, U>(a: T, b: U) -> T
where
    T: Clone + Debug,
    U: Display + Default,
{ }

// 2. Use impl Trait for simple return types
fn make_iter() -> impl Iterator<Item = i32> {
    0..10
}

// 3. Keep bounds as loose as possible
// ❌ Too restrictive
fn process<T: Display + Clone + Debug>(value: T) { }

// ✅ Looser (only needs Display)
fn process(value: &dyn Display) { }  // Or use &str for strings

// 4. Use + for multiple bounds
fn print<T: Debug + Display>(value: T) { }

// 5. Use associated types for output types
trait Iterator {
    type Item;
    fn next(&mut self) -> Option<Self::Item>;
}

// 6. Document trait bounds
/// Returns the maximum of two values.
///
/// # Type Parameters
/// * `T` - Must implement `PartialOrd` for comparison.
fn max<T: PartialOrd>(a: T, b: T) -> T { }

// 7. Use HRTB for closures that borrow
fn call_with_ref<F>(f: F) where F: for<'a> Fn(&'a i32) -> i32 { }

3️⃣ GENERICS
📖 What is it?

Generics allow writing code that works with many different types while maintaining type safety. They're resolved at compile time through monomorphization, creating zero-cost abstractions.
🎯 When to Use Generics

    Reusable data structures: Vec, Option, Result

    Reusable functions: When same logic applies to multiple types

    Algorithm implementations: Sorting, searching, mapping

    API design: Flexible interfaces

💡 Why Use Generics

    Zero-cost: No runtime overhead (monomorphization)

    Type-safe: Compiler checks type usage

    Reusable: Write once, use with many types

    Performance: Generates optimized code for each concrete type

📝 Complete Methods Reference
Generic Functions
rust

// Why generic functions: Reusable logic
// When: Same function works for multiple types

// Basic generic function
fn identity<T>(value: T) -> T {
    value
}

// Multiple type parameters
fn swap<T, U>(a: T, b: U) -> (U, T) {
    (b, a)
}

// Generic with bounds
fn max_value<T: PartialOrd>(a: T, b: T) -> T {
    if a > b { a } else { b }
}

// Generic with where clause
fn process<T, U>(value: T, transform: U) -> T
where
    T: Clone,
    U: FnOnce(T) -> T,
{
    transform(value)
}

// Usage examples
fn generic_function_examples() {
    let x = identity(42);  // T = i32
    let y = identity("hello");  // T = &str
    
    let swapped = swap(42, "hello");
    println!("{:?}", swapped);  // ("hello", 42)
    
    let max = max_value(10, 20);
    println!("{}", max);  // 20
    
    let result = process(5, |x| x * 2);
    println!("{}", result);  // 10
}

Generic Structs
rust

// Why generic structs: Flexible data containers
// When: Data structure works with multiple types

// Basic generic struct
struct Point<T> {
    x: T,
    y: T,
}

impl<T> Point<T> {
    fn new(x: T, y: T) -> Self {
        Point { x, y }
    }
    
    fn x(&self) -> &T {
        &self.x
    }
    
    fn y(&self) -> &T {
        &self.y
    }
}

// Multiple type parameters
struct Pair<T, U> {
    first: T,
    second: U,
}

impl<T, U> Pair<T, U> {
    fn new(first: T, second: U) -> Self {
        Pair { first, second }
    }
    
    fn first(&self) -> &T {
        &self.first
    }
    
    fn second(&self) -> &U {
        &self.second
    }
    
    fn swap(self) -> Pair<U, T> {
        Pair {
            first: self.second,
            second: self.first,
        }
    }
}

// Conditional implementation
impl<T: Clone> Point<T> {
    fn clone_point(&self) -> Self {
        Point {

// Why const generics: Compile-time array sizes
// When: Working with fixed-size arrays

// Array wrapper with const generic
struct ArrayWrapper<T, const N: usize> {
    data: [T; N],
}

impl<T, const N: usize> ArrayWrapper<T, N> {
    fn new(data: [T; N]) -> Self {
        ArrayWrapper { data }
    }
    
    fn len(&self) -> usize {
        N
    }
    
    fn get(&self, index: usize) -> Option<&T> {
        if index < N {
            Some(&self.data[index])
        } else {
            None
        }
    }
    
    fn map<U, F>(self, f: F) -> ArrayWrapper<U, N>
    where
        F: FnMut(T) -> U,
    {
        let mut arr = self.data;
        for item in &mut arr {
            *item = f(std::mem::take(item));
        }
        ArrayWrapper::new(arr)
    }
    
    fn iter(&self) -> std::slice::Iter<'_, T> {
        self.data.iter()
    }
}

// Fixed-size matrix
struct Matrix<T, const ROWS: usize, const COLS: usize> {
    data: [[T; COLS]; ROWS],
}

impl<T, const ROWS: usize, const COLS: usize> Matrix<T, ROWS, COLS> {
    fn new(data: [[T; COLS]; ROWS]) -> Self {
        Matrix { data }
    }
    
    fn get(&self, row: usize, col: usize) -> Option<&T> {
        if row < ROWS && col < COLS {
            Some(&self.data[row][col])
        } else {
            None
        }
    }
    
    fn get_mut(&mut self, row: usize, col: usize) -> Option<&mut T> {
        if row < ROWS && col < COLS {
            Some(&mut self.data[row][col])
        } else {
            None
        }
    }
    
    fn transpose(&self) -> Matrix<T, COLS, ROWS>
    where
        T: Copy,
    {
        let mut result = [[self.data[0][0]; ROWS]; COLS];
        for i in 0..ROWS {
            for j in 0..COLS {
                result[j][i] = self.data[i][j];
            }
        }
        Matrix::new(result)
    }
    
    fn rows(&self) -> usize {
        ROWS
    }
    
    fn cols(&self) -> usize {
        COLS
    }
}

    data: [T; N],
}

4️⃣ ASSOCIATED TYPES
📖 What is it?

Associated types are placeholders within a trait that are defined by the implementor. They make traits more readable by moving type parameters from the trait usage to the trait definition.
🎯 When to Use Associated Types

    Single output type: When each implementation has exactly one logical associated type

    Iterator patterns: Iterator::Item is the classic example

    Container traits: When the element type is determined by the container

    Cleaner APIs: When you want to avoid generic parameters in trait usage

💡 Why Use Associated Types

    Clarity: Removes generic parameters from trait usage

    Expressiveness: Makes relationships between types clear

    Flexibility: Allows traits to define associated types with lifetimes

    Simplicity: One implementation = one associated type

📝 Complete Methods Reference
Basic Associated Types

// Why associated types: Cleaner trait design
// When: Each implementation has one logical output type

struct Counter {

    max: u32,

impl Iterator for Counter {

        }

// Why multiple associated types: Complex traits
// When: Trait needs multiple output types

// Graph trait with multiple associated types
trait Graph {
    type Node;
    type Edge;
    
    fn add_node(&mut self, node: Self::Node);
    fn add_edge(&mut self, from: &Self::Node, to: &Self::Node, weight: Self::Edge);
    fn neighbors(&self, node: &Self::Node) -> Vec<&Self::Node>;
    fn edge_weight(&self, from: &Self::Node, to: &Self::Node) -> Option<&Self::Edge>;
    fn node_count(&self) -> usize;
    fn edge_count(&self) -> usize;
}

// Implementation for directed graph
use std::collections::HashMap;

struct DirectedGraph<N, E> {
    nodes: HashMap<N, Vec<(N, E)>>,
    node_list: Vec<N>,
}

impl<N, E> DirectedGraph<N, E>
where
    N: Eq + Hash + Clone,
    E: Clone,
{
    fn new() -> Self {
        DirectedGraph {
            nodes: HashMap::new(),
            node_list: Vec::new(),
        }
    }
}

impl<N, E> Graph for DirectedGraph<N, E>
where
    N: Eq + Hash + Clone,
    E: Clone,
{
    type Node = N;
    type Edge = E;
    
    fn add_node(&mut self, node: Self::Node) {
        if !self.nodes.contains_key(&node) {
            self.nodes.insert(node.clone(), Vec::new());
            self.node_list.push(node);
        }
    }
    
    fn add_edge(&mut self, from: &Self::Node, to: &Self::Node, weight: Self::Edge) {
        if let Some(edges) = self.nodes.get_mut(from) {
            edges.push((to.clone(), weight));
        }
    }
    
    fn neighbors(&self, node: &Self::Node) -> Vec<&Self::Node> {
        self.nodes
            .get(node)
            .map(|edges| edges.iter().map(|(n, _)| n).collect())
            .unwrap_or_default()
    }
    
    fn edge_weight(&self, from: &Self::Node, to: &Self::Node) -> Option<&Self::Edge> {
        self.nodes
            .get(from)?
            .iter()
            .find(|(n, _)| n == to)
            .map(|(_, w)| w)
    }
    
    fn node_count(&self) -> usize {
        self.node_list.len()
    }
    
    fn edge_count(&self) -> usize {
        self.nodes.values().map(|edges| edges.len()).sum()
    }
}

// Why GATs: Associated types with their own generic parameters
// When: Lending iterators, borrowing patterns

// Lending iterator trait (returns references that borrow from self)
trait LendingIterator {
    type Item<'a> where Self: 'a;
    
    fn next<'a>(&'a mut self) -> Option<Self::Item<'a>>;
}

// Implementation that yields references
struct Windows<'a, T> {
    slice: &'a [T],
    window_size: usize,
    position: usize,
}

impl<'a, T> LendingIterator for Windows<'a, T> {
    type Item<'b> = &'b [T] where 'a: 'b;
    
    fn next<'b>(&'b mut self) -> Option<Self::Item<'b>> {
        if self.position + self.window_size > self.slice.len() {
            return None;
        }
        
        let window = &self.slice[self.position..self.position + self.window_size];
        self.position += 1;
        Some(window)
    }
}

    fn create<'a>(&'a self) -> Self::Product<'a>;
}

impl Factory for StringFactory {
    type Product<'a> = &'a str;
    
    fn create<'a>(&'a self) -> Self::Product<'a> {
        "hello"
    }
}

5️⃣ TRAIT OBJECTS
📖 What is it?

Trait objects enable dynamic dispatch — calling methods on types determined at runtime. They're represented as dyn Trait and stored behind pointers (Box, &, Rc, Arc).
🎯 When to Use Trait Objects

    Heterogeneous collections: Storing different types in same collection

    Runtime polymorphism: When types are determined at runtime

    Binary size reduction: Avoid monomorphization bloat

    Plugin systems: Dynamically loaded implementations

    API boundaries: When you don't want to expose generic types

💡 Why Use Trait Objects

    Flexibility: Runtime polymorphism without generics

    Binary size: Single version of code instead of monomorphization

    Collections: Store different types in same collection

    Plugins: Load implementations dynamically

    Ergonomics: Simpler signatures without generic parameters

📝 Complete Methods Reference

Object Safety
rust

// Why object safety: Which traits can be trait objects
// When: Designing traits for dynamic dispatch

// ✅ Object-safe traits
trait ObjectSafe {
    fn method(&self);                    // Takes &self
    fn method_with_return(&self) -> i32; // Returns simple type
    fn method_with_ref(&self) -> &str;   // Returns reference
    fn method_with_self(&self) -> Self where Self: Sized; // Sized bound allows
}

// ❌ Not object-safe
trait NotObjectSafe {
    fn returns_self(&self) -> Self;      // Returns Self
    fn generic<T>(&self, t: T);          // Generic method
    fn consumes_self(self);               // Takes self by value
}

// ✅ Fix with where Self: Sized
trait CanBeObjectSafe {
    fn returns_self(&self) -> Self where Self: Sized;
    fn generic<T>(&self, t: T) where Self: Sized;
    fn consumes_self(self) where Self: Sized;
    
    // Object-safe methods
    fn object_safe(&self);
}

// Common object-safe traits in std
// - Debug, Display, Write
// - Fn, FnMut, FnOnce (function traits)
// - Iterator? Not object-safe due to associated type
// - Clone? Not object-safe due to returns Self

// Example: Iterator is not object-safe
// let iter: Box<dyn Iterator<Item = i32>> = Box::new(0..10);  // ❌
// But we can use boxed iterators with explicit type
let iter: Box<dyn Iterator<Item = i32>> = Box::new((0..10).map(|x| x));
// This works because we're using a concrete type that implements Iterator

trait Animal: Any {
    fn sound(&self);
    fn as_any(&self) -> &dyn Any;
}

struct Dog {
    name: String,
}

impl Animal for Dog {
    fn sound(&self) {
        println!("{} says: Woof!", self.name);
    }
    
    fn as_any(&self) -> &dyn Any {
        self
    }
}

struct Cat {
    name: String,
}

impl Animal for Cat {
    fn sound(&self) {
        println!("{} says: Meow!", self.name);
    }
    
    fn as_any(&self) -> &dyn Any {
        self
    }
}

// Downcasting (requires Any)
fn get_dog_name(animal: &dyn Animal) -> Option<&str> {
    if let Some(dog) = animal.as_any().downcast_ref::<Dog>() {
        Some(&dog.name)
    } else {
        None
    }
}

// Using downcast
fn upcast_downcast_example() {

    let cat = Cat { name: "Whiskers".to_string() };
    
    // Upcast to trait object
    let animals: Vec<&dyn Animal> = vec![&dog, &cat];
    
    for animal in animals {
        animal.sound();
        if let Some(name) = get_dog_name(animal) {
            println!("Found dog: {}", name);
        }
    }
}

Performance of Trait Objects
rust

// Why performance: Understand trade-offs
// When: Choosing between static and dynamic dispatch

// Memory layout: fat pointer (16 bytes on 64-bit)
// [data_ptr: 8 bytes] [vtable_ptr: 8 bytes]

// Vtable contains:
// - Methods (function pointers)
// - Drop implementation
// - Size and alignment info

fn performance_comparison() {
    // Static dispatch (generics) - faster, no indirection
    fn static_dispatch<T: Drawable>(shape: T) {
        shape.draw();  // Direct call (may be inlined)
    }
    
    // Dynamic dispatch (trait objects) - slower, vtable lookup
    fn dynamic_dispatch(shape: &dyn Drawable) {
        shape.draw();  // Indirect call through vtable
    }
    
    // When to use each:
    // Static dispatch:
    // - Performance-critical code
    // - Types known at compile time
    // - Small number of types
    // - Want inlining opportunities
    
    // Dynamic dispatch:
    // - Runtime flexibility needed
    // - Heterogeneous collections
    // - Reducing binary size
    // - Plugin architectures
    // - API boundaries
}

// Benchmark example (conceptual)
fn benchmark_notes() {
    // Static dispatch: 
    // - Compiler generates specialized code for each type
    // - Larger binary size
    // - Faster execution (no indirection)
    
    // Dynamic dispatch:
    // - Single version of code
    // - Smaller binary size
    // - Slower execution (vtable lookup)
    // - Cannot inline methods
}

6️⃣ OBJECT-ORIENTED PATTERNS IN RUST
📖 What is it?

Implementation of classic OOP patterns in idiomatic Rust using traits, generics, and ownership. Rust doesn't have classical inheritance but provides better alternatives through its type system.
🎯 When to Use OOP Patterns

    Encapsulation: When you need to hide implementation details

    State machines: When behavior depends on internal state

    Strategy pattern: When algorithms need to be swappable

    Factory pattern: When object creation is complex

    Observer pattern: When events need broadcasting

💡 Why Use These Patterns

    Safety: Rust's ownership prevents many OOP bugs

    Performance: Zero-cost abstractions

    Clarity: Types encode state and capabilities

    Flexibility: Traits provide interface polymorphism

Encapsulation

// Why encapsulation: Hide implementation details
// When: Protecting invariants, controlling access

mod bank {
    // Private struct - fields are private by default
    pub struct BankAccount {
        owner: String,
        account_number: String,
        balance: f64,
        transaction_history: Vec<Transaction>,
    }
    
    // Private helper struct
    struct Transaction {
        amount: f64,
        transaction_type: TransactionType,
        timestamp: u64,
    }
    
    enum TransactionType {
        Deposit,
        Withdrawal,
        Transfer,
    }
    
    impl BankAccount {
        // Public constructor
        pub fn new(owner: String, account_number: String) -> Self {
            BankAccount {
                owner,
                account_number,
                balance: 0.0,
                transaction_history: Vec::new(),
            }
        }
        
        // Public methods with validation
        pub fn deposit(&mut self, amount: f64) -> Result<(), String> {
            if amount <= 0.0 {
                return Err("Amount must be positive".to_string());
            }
            
            self.balance += amount;
            self.record_transaction(amount, TransactionType::Deposit);
            Ok(())
        }
        
        pub fn withdraw(&mut self, amount: f64) -> Result<(), String> {
            if amount <= 0.0 {
                return Err("Amount must be positive".to_string());
            }
            
            if amount > self.balance {
                return Err("Insufficient funds".to_string());
            }
            
            self.balance -= amount;
            self.record_transaction(amount, TransactionType::Withdrawal);
            Ok(())
        }
        
        pub fn transfer(&mut self, to: &mut BankAccount, amount: f64) -> Result<(), String> {
            self.withdraw(amount)?;
            to.deposit(amount)?;
            self.record_transaction(amount, TransactionType::Transfer);
            Ok(())
        }
        
        // Read-only access
        pub fn balance(&self) -> f64 {
            self.balance
        }
        
        pub fn owner(&self) -> &str {
            &self.owner
        }
        
        pub fn account_number(&self) -> &str {
            &self.account_number
        }
        
        // Private helper
        fn record_transaction(&mut self, amount: f64, tx_type: TransactionType) {
            self.transaction_history.push(Transaction {
                amount,
                transaction_type: tx_type,
                timestamp: std::time::SystemTime::now()
                    .duration_since(std::time::UNIX_EPOCH)
                    .unwrap()
                    .as_secs(),
            });
        }
    }
}

Strategy Pattern
rust

// Why strategy: Swap algorithms at runtime
// When: Multiple algorithms for same task

// Strategy trait
trait CompressionStrategy {
    fn compress(&self, data: &[u8]) -> Vec<u8>;
    fn decompress(&self, data: &[u8]) -> Vec<u8>;
    fn name(&self) -> &str;
}

// Concrete strategies
struct ZipCompression;

impl CompressionStrategy for ZipCompression {
    fn compress(&self, data: &[u8]) -> Vec<u8> {
        println!("Compressing with ZIP");
        // Simulated compression
        data.iter().map(|&b| b ^ 0xFF).collect()
    }
    
    fn decompress(&self, data: &[u8]) -> Vec<u8> {
        println!("Decompressing with ZIP");
        data.iter().map(|&b| b ^ 0xFF).collect()
    }
    
    fn name(&self) -> &str {
        "ZIP"
    }
}

struct GzipCompression;

impl CompressionStrategy for GzipCompression {
    fn compress(&self, data: &[u8]) -> Vec<u8> {
        println!("Compressing with GZIP");
        data.iter().map(|&b| b.wrapping_add(1)).collect()
    }
    
    fn decompress(&self, data: &[u8]) -> Vec<u8> {
        println!("Decompressing with GZIP");
        data.iter().map(|&b| b.wrapping_sub(1)).collect()
    }
    
    fn name(&self) -> &str {
        "GZIP"
    }
}

// Context that uses the strategy
struct Compressor {
    strategy: Box<dyn CompressionStrategy>,
}

impl Compressor {
    fn new(strategy: Box<dyn CompressionStrategy>) -> Self {
        Compressor { strategy }
    }
    
    fn set_strategy(&mut self, strategy: Box<dyn CompressionStrategy>) {
        self.strategy = strategy;
    }
    
    fn compress(&self, data: &[u8]) -> Vec<u8> {
        println!("Using strategy: {}", self.strategy.name());
        self.strategy.compress(data)
    }
    
    fn decompress(&self, data: &[u8]) -> Vec<u8> {
        self.strategy.decompress(data)
    }
}

// Functional approach (with closures)
struct FunctionalCompressor;

impl FunctionalCompressor {
    fn compress<F>(data: &[u8], strategy: F) -> Vec<u8>
    where
        F: Fn(&[u8]) -> Vec<u8>,
    {
        strategy(data)
    }
}

// Usage
fn strategy_pattern_example() {
    let data = b"Hello, World!";
    
    let mut compressor = Compressor::new(Box::new(ZipCompression));
    let compressed = compressor.compress(data);
    println!("Compressed: {:?}", compressed);
    
    compressor.set_strategy(Box::new(GzipCompression));
    let compressed2 = compressor.compress(data);
    println!("Compressed: {:?}", compressed2);
    
    // Functional approach
    let zip_fn = |data: &[u8]| data.iter().map(|&b| b ^ 0xFF).collect();
    let result = FunctionalCompressor::compress(data, zip_fn);
    println!("Functional: {:?}", result);
}

// Traditional state pattern (runtime)
enum ConnectionState {
    Disconnected,
    Connecting,
    Connected,
    Error(String),
}

struct Connection {
    state: ConnectionState,
}

impl Connection {
    fn new() -> Self {
        Connection {
            state: ConnectionState::Disconnected,
        }
    }
    
    fn connect(&mut self) -> Result<(), String> {
        match &self.state {
            ConnectionState::Disconnected => {
                println!("Connecting...");
                self.state = ConnectionState::Connecting;
                // Simulate connection
                self.state = ConnectionState::Connected;
                Ok(())
            }
            ConnectionState::Connected => {
                Err("Already connected".to_string())
            }
            ConnectionState::Connecting => {
                Err("Already connecting".to_string())
            }
            ConnectionState::Error(e) => {
                Err(format!("Cannot connect: {}", e))
            }
        }
    }
    
    fn disconnect(&mut self) -> Result<(), String> {
        match &self.state {
            ConnectionState::Connected => {
                println!("Disconnecting...");
                self.state = ConnectionState::Disconnected;
                Ok(())
            }
            _ => Err("Not connected".to_string()),
        }
    }
    
    fn send(&self, data: &[u8]) -> Result<(), String> {
        match &self.state {
            ConnectionState::Connected => {
                println!("Sending: {:?}", data);
                Ok(())
            }
            _ => Err("Not connected".to_string()),
        }
    }
}

Observer Pattern
rust

// Why observer: Event broadcasting
// When: Multiple listeners need notification

use std::rc::Rc;
use std::cell::RefCell;

type ObserverCallback = Box<dyn Fn(&str)>;

struct Subject {
    observers: Vec<Rc<RefCell<ObserverCallback>>>,
    state: String,
}

impl Subject {
    fn new() -> Self {
        Subject {
            observers: Vec::new(),
            state: String::new(),
        }
    }
    
    fn attach(&mut self, observer: Rc<RefCell<ObserverCallback>>) {
        self.observers.push(observer);
    }
    
    fn detach(&mut self, observer: Rc<RefCell<ObserverCallback>>) {
        self.observers.retain(|o| !Rc::ptr_eq(o, &observer));
    }
    
    fn set_state(&mut self, state: String) {
        self.state = state;
        self.notify_observers();
    }
    
    fn notify_observers(&self) {
        for observer in &self.observers {
            let callback = observer.borrow();
            callback(&self.state);
        }
    }
}

struct Observer {
    name: String,
}

impl Observer {
    fn new(name: String) -> Self {
        Observer { name }
    }
    
    fn create_callback(&self) -> Rc<RefCell<ObserverCallback>> {
        let name = self.name.clone();
        Rc::new(RefCell::new(Box::new(move |state| {
            println!("[{}] Received: {}", name, state);
        }) as ObserverCallback))
    }
}

// Functional observer pattern (using channels)
use std::sync::mpsc;
use std::thread;

struct EventEmitter {
    tx: mpsc::Sender<String>,
    rx: mpsc::Receiver<String>,
}

impl EventEmitter {
    fn new() -> Self {
        let (tx, rx) = mpsc::channel();
        EventEmitter { tx, rx }
    }
    
    fn on<F>(&self, callback: F) -> thread::JoinHandle<()>
    where
        F: Fn(String) + Send + 'static,
    {
        let rx = self.rx.try_clone().unwrap();
        thread::spawn(move || {
            while let Ok(event) = rx.recv() {
                callback(event);
            }
        })
    }
    
    fn emit(&self, event: String) {
        self.tx.send(event).unwrap();
    }
}

// Usage
fn observer_pattern_example() {
    // Traditional observer
    let mut subject = Subject::new();
    
    let observer1 = Observer::new("Logger".to_string());
    let observer2 = Observer::new("Notifier".to_string());
    
    let callback1 = observer1.create_callback();
    let callback2 = observer2.create_callback();
    
    subject.attach(callback1);
    subject.attach(callback2);
    
    subject.set_state("Application started".to_string());
    subject.set_state("User logged in".to_string());
    
    // Functional observer with channels
    let emitter = EventEmitter::new();
    
    let handle1 = emitter.on(|event| {
        println!("Handler 1: {}", event);
    });
    
    let handle2 = emitter.on(|event| {
        println!("Handler 2: {}", event);
    });
    
    emitter.emit("Custom event".to_string());
    emitter.emit("Another event".to_string());
    
    // Clean up
    drop(emitter);
    handle1.join().unwrap();
    handle2.join().unwrap();
}
