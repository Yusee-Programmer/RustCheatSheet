1️⃣ RUST BEST PRACTICES
📖 What is it?

Rust best practices are conventions and patterns that make code idiomatic, maintainable, and performant. Following these practices ensures your code is readable, consistent, and follows the Rust community's collective wisdom.
🎯 When to Apply Best Practices

    Always: In all Rust code you write, regardless of project size

    Code reviews: When reviewing others' code to maintain consistency

    Library design: When creating public APIs that others will use

    Onboarding: When teaching Rust to new team members

    Refactoring: When improving existing codebases

💡 Why Follow Best Practices

    Consistency: Code looks similar across projects, reducing cognitive load

    Readability: Easier for others (and future you) to understand

    Maintainability: Easier to modify, extend, and debug

    Performance: Often leads to better performance through idiomatic patterns

    Safety: Avoids common pitfalls and edge cases

    Community: Aligns with ecosystem expectations

Naming Conventions
rust

// ✅ GOOD: Snake case for variables and functions
let user_name = "Alice";
let user_age = 30;
fn calculate_total() -> i32 { 42 }
fn process_user_data() { }

// ❌ BAD: Camel case in Rust (reserved for types)
let userName = "Alice";
fn CalculateTotal() -> i32 { 42 }

// ✅ GOOD: Camel case for types (UpperCamelCase)
struct UserProfile;
enum ColorType;
trait Serializable;
type Result<T> = std::result::Result<T, MyError>;

// ❌ BAD: Snake case for types
struct user_profile;
enum color_type;

// ✅ GOOD: SCREAMING_SNAKE for constants and statics
const MAX_CONNECTIONS: u32 = 100;
const DEFAULT_TIMEOUT_MS: u64 = 5000;
static APPLICATION_NAME: &str = "MyApp";

// ✅ GOOD: UpperCamelCase for enum variants
enum HttpStatus {
    Ok,                    // Not OK (capitalized)
    NotFound,
    InternalServerError,
}

// ✅ GOOD: snake_case for module names and file names
mod user_authentication;
mod database_connection;

// File: user_authentication.rs
// File: database_connection.rs

// ✅ GOOD: snake_case for macro names
macro_rules! my_macro {
    () => {};
}

// ✅ GOOD: Prefixed underscores for unused variables
let _unused = 42;
fn process(_arg: i32) { }

// ✅ GOOD: Single-character for simple loops
for i in 0..10 { }  // i is fine for loop index
let x = 5;          // x is fine for simple values

// ❌ BAD: Single-character for complex meanings
let a = calculate_something();  // What is 'a'?

Error Handling Best Practices
rust

// ✅ GOOD: Use Result for recoverable errors
fn read_config(path: &str) -> Result<Config, ConfigError> {
    let content = std::fs::read_to_string(path)?;
    serde_json::from_str(&content).map_err(ConfigError::Parse)
}

// ✅ GOOD: Create custom error types for libraries using thiserror
use thiserror::Error;

#[derive(Error, Debug)]
pub enum ApiError {
    #[error("Network error: {0}")]
    Network(#[from] std::io::Error),
    
    #[error("Parse error: {0}")]
    Parse(#[from] serde_json::Error),
    
    #[error("Invalid input: {0}")]
    InvalidInput(String),
    
    #[error("Not found: {0}")]
    NotFound(String),
    
    #[error("Unauthorized")]
    Unauthorized,
}

// ✅ GOOD: Use anyhow for application error handling
use anyhow::{Context, Result};

fn main() -> Result<()> {
    let config = read_config("config.json")
        .context("Failed to read config file")?;
    
    let data = fetch_data(&config)
        .context("Failed to fetch data from API")?;
    
    process(data).context("Failed to process data")?;
    
    Ok(())
}

// ✅ GOOD: Use expect for invariants that should never fail
let config = read_config("config.json")
    .expect("Config file is required and must be valid at startup");

// ✅ GOOD: Use unwrap only in tests and prototypes
#[cfg(test)]
mod tests {
    #[test]
    fn test_parse() {
        let value = "42".parse::<i32>().unwrap();  // OK in tests
        assert_eq!(value, 42);
    }
}

// ❌ BAD: Using unwrap in production code
let data = read_file().unwrap();  // May panic in production!

// ✅ GOOD: Add context to errors for better debugging
fn parse_config(path: &str) -> Result<Config> {
    let content = std::fs::read_to_string(path)
        .with_context(|| format!("Failed to read config file: {}", path))?;
    
    serde_json::from_str(&content)
        .with_context(|| format!("Failed to parse JSON from {}", path))
}

// ✅ GOOD: Use Box<dyn Error> for simple applications
use std::error::Error;

fn main() -> Result<(), Box<dyn Error>> {
    let data = std::fs::read_to_string("data.txt")?;
    let num: i32 = data.trim().parse()?;
    println!("Number: {}", num);
    Ok(())
}

// ✅ GOOD: Use Option for optional values (not errors)
fn find_user(id: u32) -> Option<User> {

// ✅ GOOD: Use match for exhaustive handling

    None => (),

// ✅ GOOD: Use @ for binding

    n @ 1..=10 => println!("Small number: {}", n),
    n @ 20..=30 => println!("Medium number: {}", n),
    n => println!("Large number: {}", n),
}

// ✅ GOOD: Use guard clauses for additional conditions
match number {
    n if n % 2 == 0 => println!("Even: {}", n),
    n if n % 2 != 0 => println!("Odd: {}", n),
    _ => unreachable!(),
}

Ownership and Borrowing Best Practices
rust

// ✅ GOOD: Use references when you don't need ownership
fn process_data(data: &Data) -> Result<(), Error> {
    // Only reads data, no need to own
    println!("Processing: {}", data.name);
    Ok(())
}

// ✅ GOOD: Use &mut when you need to modify
fn update_data(data: &mut Data) {
    data.value += 1;
    data.updated_at = Utc::now();
}

// ✅ GOOD: Use move closures when transferring ownership
let data = vec![1, 2, 3];
thread::spawn(move || {
    process_data(data);  // Data moved into thread
});

// ✅ GOOD: Use Cow for either owned or borrowed data
use std::borrow::Cow;

fn to_uppercase(s: &str) -> Cow<'_, str> {
    if s.chars().any(|c| c.is_lowercase()) {
        Cow::Owned(s.to_uppercase())  // Need to allocate
    } else {
        Cow::Borrowed(s)  // No allocation needed
    }
}

// ✅ GOOD: Use Clone sparingly; prefer references when possible
fn process_users(users: &[User]) {  // Borrow, not clone
    for user in users {
        process(user);  // Process borrows, not clones
    }
}

// ✅ GOOD: Use .clone() only when you truly need independent ownership
fn store_user(user: &User) {
    let user_owned = user.clone();  // Need owned copy for storage
    DATABASE.insert(user_owned);
}

// ✅ GOOD: Use .to_owned() for &str → String conversion
let s: String = "hello".to_owned();

Generics and Traits Best Practices
rust

// ✅ GOOD: Use where clause for complex bounds (more readable)
fn process<T, U>(a: T, b: U) -> T
where
    T: Clone + Debug,
    U: Display + Default,
{
    println!("a: {:?}", a);
    println!("b: {}", b);
    a.clone()
}

// ❌ BAD: Inline bounds for complex constraints (hard to read)
fn process<T: Clone + Debug, U: Display + Default>(a: T, b: U) -> T { }

// ✅ GOOD: Use impl Trait for simple return types
fn make_iter() -> impl Iterator<Item = i32> {
    0..10
}

// ✅ GOOD: Use trait objects for runtime polymorphism
let shapes: Vec<Box<dyn Shape>> = vec![
    Box::new(Circle { radius: 1.0 }),
    Box::new(Square { side: 2.0 }),
];

// ✅ GOOD: Use associated types for single output types
trait Iterator {
    type Item;  // Each iterator has one Item type
    fn next(&mut self) -> Option<Self::Item>;
}

// ✅ GOOD: Use generics for multiple implementations
trait From<T> {  // Can be implemented multiple times for same type
    fn from(value: T) -> Self;
}

// ✅ GOOD: Use supertraits for trait hierarchies
trait Animal {
    fn name(&self) -> &str;
}

trait Pet: Animal {  // Pet requires Animal
    fn owner(&self) -> &str;
}

// ✅ GOOD: Provide default implementations when possible
trait Logger {
    fn log(&self, msg: &str);
    
    // Default implementations
    fn error(&self, msg: &str) {
        self.log(&format!("ERROR: {}", msg));
    }
    
    fn warn(&self, msg: &str) {
        self.log(&format!("WARN: {}", msg));
    }
}

///
/// # Examples
///
/// ```
/// use geometry::distance;
///

2️⃣ CODE ORGANIZATION
📖 What is it?

Code organization in Rust involves structuring your project into modules, crates, and workspaces to maintain clarity, separation of concerns, and reusability as your codebase grows.
🎯 When to Apply Organization

    Small projects (1-5 files): Use single crate with modules

    Medium projects (5-20 files): Split into modules with clear boundaries

    Large projects (20+ files): Use multiple crates in a workspace

    Libraries: Create focused, reusable crates with minimal dependencies

    Applications: Separate concerns into distinct layers

💡 Why Organize Code

    Maintainability: Easier to find and modify code

    Separation of concerns: Each module has a single responsibility

    Reusability: Code can be shared across projects

    Compilation speed: Only recompile changed modules

    Team collaboration: Clear ownership of code areas

    Testing: Isolated modules are easier to test

Separation of Concerns
rust

// ✅ GOOD: Separate models from business logic
// models/user.rs - Pure data structures
#[derive(Debug, Serialize, Deserialize)]
pub struct User {
    pub id: UserId,
    pub name: String,
    pub email: String,
}

// services/user_service.rs - Business logic
pub struct UserService {
    repository: Box<dyn UserRepository>,
    validator: Box<dyn UserValidator>,
}

impl UserService {
    pub async fn create_user(&self, name: String, email: String) -> Result<User> {
        // Validation
        self.validator.validate_name(&name)?;
        self.validator.validate_email(&email)?;
        
        // Business logic
        let user = User::new(name, email);
        
        // Persistence
        self.repository.save(&user).await?;
        
        Ok(user)
    }
}

// repositories/user_repository.rs - Data access
#[async_trait]
pub trait UserRepository: Send + Sync {
    async fn find_by_id(&self, id: UserId) -> Result<Option<User>>;
    async fn find_by_email(&self, email: &str) -> Result<Option<User>>;
    async fn save(&self, user: &User) -> Result<User>;
}

// handlers/user_handler.rs - HTTP layer
pub async fn create_user(
    State(service): State<Arc<UserService>>,
    Json(payload): Json<CreateUserRequest>,
) -> impl IntoResponse {
    match service.create_user(payload.name, payload.email).await {
        Ok(user) => (StatusCode::CREATED, Json(user)).into_response(),
        Err(e) => handle_error(e).into_response(),
    }
}

3️⃣ PERFORMANCE OPTIMIZATION
📖 What is it?

Performance optimization in Rust involves identifying bottlenecks, using profiling tools, and applying optimization techniques to make your code faster and more memory-efficient.
🎯 When to Optimize

    After profiling: Only optimize what's actually slow (measure first!)

    Hot paths: Critical sections that run frequently (inner loops)

    Memory constraints: When memory usage is high or growing

    Startup time: For CLI tools and services where startup latency matters

    Latency-sensitive: Real-time applications, user-facing services

    Throughput: When handling high request volumes

💡 Why Optimize

// benches/benchmark.rs
use criterion::{criterion_group, criterion_main, Criterion, black_box};

fn fibonacci_recursive(n: u64) -> u64 {
    match n {
        0 => 0,
        1 => 1,
        _ => fibonacci_recursive(n - 1) + fibonacci_recursive(n - 2),
    }
}

fn fibonacci_iterative(n: u64) -> u64 {
    let mut a = 0;
    let mut b = 1;
    for _ in 0..n {
        let c = a + b;
        a = b;
        b = c;
    }
    a
}

fn bench_fibonacci(c: &mut Criterion) {
    let mut group = c.benchmark_group("Fibonacci");
    
    // Compare different implementations
    group.bench_function("recursive 30", |b| {
        b.iter(|| fibonacci_recursive(black_box(30)))
    });
    
    group.bench_function("iterative 30", |b| {
        b.iter(|| fibonacci_iterative(black_box(30)))
    });
    
    // Parameterized benchmarks
    for size in [10, 20, 30, 40].iter() {
        group.bench_with_input(
            format!("recursive_{}", size),
            size,
            |b, &size| b.iter(|| fibonacci_recursive(black_box(size)))
        );
    }
    
    group.finish();
}

// ✅ GOOD: Choose appropriate data structures
// HashMap: O(1) lookups for key-value
// BTreeMap: O(log n) lookups with sorted order

        .iter()

use tokio::sync::Semaphore;

async fn process_many(items: &[Item]) {
    let semaphore = Semaphore::new(10);  // Max 10 concurrent
    
    let mut handles = vec![];
    for item in items {
        let permit = semaphore.acquire().await.unwrap();
        let handle = tokio::spawn(async move {
            process(item).await;
            drop(permit);
        });
        handles.push(handle);
    }
    
    futures::future::join_all(handles).await;
}

    pub database: DatabaseConfig,
    pub redis: RedisConfig,
    pub logging: LoggingConfig,

#[derive(Debug, Clone, Deserialize, PartialEq)]
#[serde(rename_all = "lowercase")]
pub enum Environment {
    Development,
    Staging,
    Production,
}

impl Config {
    pub fn from_env() -> Result<Self, ConfigError> {
        // Try to load from file first
        let config = if let Ok(path) = env::var("CONFIG_PATH") {
            Self::from_file(&path)?
        } else {
            Self::from_env_vars()?
        };
        
        config.validate()?;
        Ok(config)
    }
    
    fn from_file(path: &str) -> Result<Self, ConfigError> {
        let content = std::fs::read_to_string(path)?;
        Ok(serde_yaml::from_str(&content)?)
    }
    
    fn from_env_vars() -> Result<Self, ConfigError> {
        Ok(Config {
            environment: env::var("ENVIRONMENT")
                .unwrap_or_else(|_| "development".into())
                .parse()?,
            server: ServerConfig {
                host: env::var("SERVER_HOST").unwrap_or_else(|_| "0.0.0.0".into()),
                port: env::var("SERVER_PORT").unwrap_or_else(|_| "8080".into()).parse()?,
                workers: env::var("SERVER_WORKERS")
                    .unwrap_or_else(|_| num_cpus::get().to_string())
                    .parse()?,
                timeout_secs: env::var("SERVER_TIMEOUT")
                    .unwrap_or_else(|_| "30".into())
                    .parse()?,
            },
            database: DatabaseConfig {
                url: env::var("DATABASE_URL")?,
                max_connections: env::var("DATABASE_POOL_SIZE")
                    .unwrap_or_else(|_| "10".into())
                    .parse()?,
                connect_timeout_secs: env::var("DATABASE_CONNECT_TIMEOUT")
                    .unwrap_or_else(|_| "10".into())
                    .parse()?,
            },
            redis: RedisConfig {
                url: env::var("REDIS_URL").unwrap_or_else(|_| "redis://localhost".into()),
                max_connections: env::var("REDIS_POOL_SIZE")
                    .unwrap_or_else(|_| "5".into())
                    .parse()?,
            },
            logging: LoggingConfig {
                level: env::var("LOG_LEVEL").unwrap_or_else(|_| "info".into()),
                format: env::var("LOG_FORMAT").unwrap_or_else(|_| "json".into()),
            },
        })
    }
    
    fn validate(&self) -> Result<(), ConfigError> {
        if self.server.port == 0 || self.server.port > 65535 {
            return Err(ConfigError::InvalidPort);
        }
        if self.database.max_connections == 0 {
            return Err(ConfigError::InvalidPoolSize);
        }
        if self.server.workers == 0 {
            return Err(ConfigError::InvalidWorkers);
        }
        Ok(())
    }
}

// src/services/user_service.rs - Business logic layer
use crate::{
    models::{User, UserId},
    repositories::UserRepository,
    error::{Error, Result},
};
use tracing::{info, warn};

pub struct UserService {
    repository: Box<dyn UserRepository>,
}

impl UserService {
    pub fn new(repository: Box<dyn UserRepository>) -> Self {
        UserService { repository }
    }
    
    pub async fn get_user(&self, id: UserId) -> Result<Option<User>> {
        info!("Getting user: {:?}", id);
        self.repository.find_by_id(id).await
    }
    
    pub async fn create_user(&self, name: String, email: String) -> Result<User> {
        info!("Creating user: {}", email);
        
        // Validate input
        if name.is_empty() {
            return Err(Error::Validation("Name cannot be empty".into()));
        }
        
        if !email.contains('@') || !email.contains('.') {
            return Err(Error::Validation("Invalid email format".into()));
        }
        
        // Check for existing user
        if let Some(existing) = self.repository.find_by_email(&email).await? {
            warn!("User already exists: {}", email);
            return Err(Error::Conflict("User already exists".into()));
        }
        
        // Create user
        let user = User::new(name, email);
        let created = self.repository.create(&user).await?;
        
        info!("User created: {:?}", created.id);
        Ok(created)
    }
    
    pub async fn update_email(&self, id: UserId, email: String) -> Result<User> {
        info!("Updating email for user: {:?}", id);
        
        // Validate
        if !email.contains('@') {
            return Err(Error::Validation("Invalid email format".into()));
        }
        
        // Get user
        let mut user = self.repository
            .find_by_id(id)
            .await?
            .ok_or_else(|| Error::NotFound(format!("User {} not found", id.0)))?;
        
        // Update
        user.update_email(email);
        
        // Save
        let updated = self.repository.update(&user).await?;
        
        info!("Email updated for user: {:?}", id);
        Ok(updated)
    }
    
    pub async fn delete_user(&self, id: UserId) -> Result<()> {
        info!("Deleting user: {:?}", id);
        
        // Check exists
        if self.repository.find_by_id(id).await?.is_none() {
            return Err(Error::NotFound(format!("User {} not found", id.0)));
        }
        
        self.repository.delete(id).await?;
        
        info!("User deleted: {:?}", id);
        Ok(())
    }
}

use axum::{
    extract::{Path, State},
    Json,
    response::{IntoResponse, Response},
    http::StatusCode,
};
use crate::{
    services::UserService,
    models::{User, UserId},
    error::Error,
};
use tracing::error;

pub async fn get_user(
    State(service): State<axum::extract::Extension<Arc<UserService>>>,
    Path(id): Path<i64>,
) -> impl IntoResponse {
    match service.get_user(UserId(id)).await {
        Ok(Some(user)) => (StatusCode::OK, Json(user)).into_response(),
        Ok(None) => (StatusCode::NOT_FOUND, "User not found").into_response(),
        Err(e) => handle_error(e),
    }
}

pub async fn create_user(
    State(service): State<axum::extract::Extension<Arc<UserService>>>,
    Json(payload): Json<CreateUserRequest>,
) -> impl IntoResponse {
    match service.create_user(payload.name, payload.email).await {
        Ok(user) => (StatusCode::CREATED, Json(user)).into_response(),
        Err(e) => handle_error(e),
    }
}

pub async fn update_user_email(
    State(service): State<axum::extract::Extension<Arc<UserService>>>,
    Path(id): Path<i64>,
    Json(payload): Json<UpdateEmailRequest>,
) -> impl IntoResponse {
    match service.update_email(UserId(id), payload.email).await {
        Ok(user) => (StatusCode::OK, Json(user)).into_response(),
        Err(e) => handle_error(e),
    }
}

#[derive(Debug, Deserialize)]
pub struct CreateUserRequest {
    name: String,
    email: String,
}

#[derive(Debug, Deserialize)]
pub struct UpdateEmailRequest {
    email: String,
}

fn handle_error(error: Error) -> Response {
    error!("Request failed: {}", error);
    
    let (status, message) = match error {
        Error::Validation(msg) => (StatusCode::BAD_REQUEST, msg),
        Error::NotFound(msg) => (StatusCode::NOT_FOUND, msg),
        Error::Conflict(msg) => (StatusCode::CONFLICT, msg),
        Error::Unauthorized => (StatusCode::UNAUTHORIZED, "Unauthorized".into()),
        _ => (StatusCode::INTERNAL_SERVER_ERROR, "Internal error".into()),
    };
    
    (status, message).into_response()
}

use axum::{
    Router,

use std::sync::Arc;
use tower_http::trace::TraceLayer;

pub async fn run(config: Config) -> anyhow::Result<()> {
    // Initialize database pool
    let pool = init_database(&config.database).await?;
    
    // Initialize repositories
    let user_repo = Arc::new(PostgresUserRepository::new(pool));

    let user_service = Arc::new(UserService::new(user_repo));

    let app = Router::new()
        .route("/users", post(handlers::create_user))
        .route("/users/:id", get(handlers::get_user))
        .route("/users/:id/email", put(handlers::update_user_email))
        .route("/users/:id", delete(handlers::delete_user))
        .layer(Extension(user_service))
        .layer(Extension(config.clone()))
        .layer(middleware::from_fn(middleware::logging))
        .layer(middleware::from_fn(middleware::metrics))
        .layer(TraceLayer::new_for_http());

    let addr = std::net::SocketAddr::from(([0, 0, 0, 0], config.server.port));
    tracing::info!("Server listening on {}", addr);

        .serve(app.into_make_service())
        .with_graceful_shutdown(shutdown_signal())
        .await?;
    
    Ok(())
}

async fn shutdown_signal() {
    let ctrl_c = async {
        tokio::signal::ctrl_c()
            .await
            .expect("failed to install Ctrl+C handler");
        tracing::info!("Ctrl+C received, shutting down...");
    };
    
    #[cfg(unix)]
    let terminate = async {
        tokio::signal::unix::signal(tokio::signal::unix::SignalKind::terminate())
            .expect("failed to install signal handler")
            .recv()
            .await;
        tracing::info!("SIGTERM received, shutting down...");
    };
    
    #[cfg(not(unix))]
    let terminate = std::future::pending::<()>();
    
    tokio::select! {
        _ = ctrl_c => {},
        _ = terminate => {},
    }
}

// src/handlers/health.rs
use axum::{Json, response::IntoResponse};
use serde::Serialize;
use std::time::{SystemTime, UNIX_EPOCH};

#[derive(Serialize)]
struct HealthResponse {
    status: &'static str,
    version: &'static str,
    timestamp: u64,
    uptime: u64,
}

static START_TIME: std::sync::OnceLock<u64> = std::sync::OnceLock::new();

pub async fn health() -> impl IntoResponse {
    let start = *START_TIME.get_or_init(|| {
        SystemTime::now()
            .duration_since(UNIX_EPOCH)
            .unwrap()
            .as_secs()
    });
    
    let now = SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap()
        .as_secs();
    
    Json(HealthResponse {
        status: "ok",
        version: env!("CARGO_PKG_VERSION"),
        timestamp: now,
        uptime: now - start,
    })
}
